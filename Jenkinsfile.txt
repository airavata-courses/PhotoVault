pipeline {
    agent any

    stages {
        stage('Build React App') {
            steps {
                sh 'sudo apt-get install nodejs-legacy'
                sh 'sudo apt-get install npm'
                dir('website'){
                   sh 'npm install'
                }
            }
        }
        
        stage('Deploy') {
			steps {
				sh '''
                    	kill -9 $(lsof -i:3000 -t) || echo $?
						npm install
						sudo npm run build
						sudo npm install -g serve
						BUILD_ID=dontKillMe nohup serve -s build -l 3000 &
				'''
				
			}
		}
    }
}
